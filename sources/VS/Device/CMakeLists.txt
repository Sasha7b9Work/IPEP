file(GLOB DISPLAY_FONT          ../../Device/src/Display/Font/*.*)
file(GLOB HARDWARE              ../../Device/src/Hardware/*.*)

                                                 Hardware/HAL/*.*)

                ${LIBS_STM32F1}

                ${DISPLAY}
                ${DISPLAY_FONT}
                ${HARDWARE}
                ${HARDWARE_CDC}
                ${HARDWARE_HAL}
                ${HARDWARE_I2C}
                ${HARDWARE_LAN}
                ${HARDWARE_LAN_LIBNET}
